\subsection{MPG auto v manual}

\begin{Schunk}
\begin{Sinput}
> autompg <- mean (autocars$mpg)
> mancars <- mtcars[mtcars$am==0,]
> manmpg <- mean (mancars$mpg)
> cat("Auto MPG: \t", autompg)
\end{Sinput}
\begin{Soutput}
Auto MPG: 	 24.39231
\end{Soutput}
\begin{Sinput}
> cat("Man MPG: \t", manmpg)
\end{Sinput}
\begin{Soutput}
Man MPG: 	 17.14737
\end{Soutput}
\end{Schunk}
